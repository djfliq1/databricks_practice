# notebooks/01_bronze_ingestion.py
# Enhanced version of your 2_bronze_PySpark notebook

# Databricks notebook source
# MAGIC %md
# MAGIC ## Bronze Layer Ingestion - Products & Customers
# MAGIC **Part of LakeFlow Jobs Medallion Pipeline**

# COMMAND ----------
# Import libraries
from pyspark.sql.functions import *
from pyspark.sql.types import *
from delta.tables import DeltaTable
import datetime

# COMMAND ----------
# MAGIC %md
# MAGIC ### Configure Job Parameters

# COMMAND ----------
# Widget parameters for job flexibility
dbutils.widgets.text("catalog_name", "databricks_clar", "Unity Catalog Name")
dbutils.widgets.text("schema_name", "bronze", "Schema Name")
dbutils.widgets.text("source_path", "/Volumes/databricks_clar/bronze/bronze_volume/", "Source Data Path")

catalog_name = dbutils.widgets.get("catalog_name")
schema_name = dbutils.widgets.get("schema_name")
source_path = dbutils.widgets.get("source_path")

print(f"Processing data for {catalog_name}.{schema_name}")

# COMMAND ----------
# MAGIC %md
# MAGIC ### Products Processing (From Your CSV)

# COMMAND ----------
# Load products data (your original logic enhanced)
df_products = spark.read.format("csv") \
  .option("header", True) \
  .option("inferSchema", True) \
  .load(f"{source_path}products/")

# Add audit columns for data lineage
df_products_enhanced = df_products.select(
    col("product_id"),
    col("product_name"),
    col("category"),
    col("price"),
    col("processDate").alias("source_process_date"),
    current_timestamp().alias("bronze_ingestion_timestamp"),
    lit("products_csv").alias("source_system"),
    hash(concat_ws("||", *df_products.columns)).alias("source_hash")
)

# Data quality metrics for downstream validation
product_row_count = df_products_enhanced.count()
product_null_count = df_products_enhanced.filter(
    col("product_id").isNull() | 
    col("product_name").isNull() | 
    col("category").isNull()
).count()

print(f"Products processed: {product_row_count} rows, {product_null_count} nulls")

# COMMAND ----------
# MAGIC %md
# MAGIC ### Customers Processing

# COMMAND ----------
# Load customers data (your original logic)
df_customers = spark.read.format("csv") \
  .option("header", True) \
  .option("inferSchema", True) \
  .load(f"{source_path}customers/")

# Extract domain from email for your domain analysis
df_customers_enhanced = df_customers.select(
    col("customer_id"),
    col("name"),
    col("email"),
    col("location"),
    col("signup_date"),
    regexp_extract(col("email"), "@(.+)", 1).alias("domain"),
    current_timestamp().alias("bronze_ingestion_timestamp"),
    lit("customers_csv").alias("source_system"),
    hash(concat_ws("||", *df_customers.columns)).alias("source_hash")
)

customer_row_count = df_customers_enhanced.count()
customer_null_count = df_customers_enhanced.filter(
    col("customer_id").isNull() | 
    col("email").isNull()
).count()

print(f"Customers processed: {customer_row_count} rows, {customer_null_count} nulls")

# COMMAND ----------
# MAGIC %md
# MAGIC ### Write to Bronze Layer (Delta Tables)

# COMMAND ----------
# Write products to Bronze
products_table = f"{catalog_name}.{schema_name}.products_bronze"

# Your original upsert logic enhanced
if spark.catalog.tableExists(products_table):
    dlt_products = DeltaTable.forName(spark, products_table)
    dlt_products.alias("trg") \
        .merge(df_products_enhanced.alias("src"), "trg.product_id = src.product_id") \
        .whenMatchedUpdateAll() \
        .whenNotMatchedInsertAll() \
        .execute()
    print(f"✅ Merged {product_row_count} products into {products_table}")
else:
    df_products_enhanced.write.format("delta") \
        .mode("append") \
        .saveAsTable(products_table)
    print(f"✅ Created {products_table} with {product_row_count} products")

# COMMAND ----------
# Write customers to Bronze
customers_table = f"{catalog_name}.{schema_name}.customers_bronze"

if spark.catalog.tableExists(customers_table):
    dlt_customers = DeltaTable.forName(spark, customers_table)
    dlt_customers.alias("trg") \
        .merge(df_customers_enhanced.alias("src"), "trg.customer_id = src.customer_id") \
        .whenMatchedUpdateAll() \
        .whenNotMatchedInsertAll() \
        .execute()
    print(f"✅ Merged {customer_row_count} customers into {customers_table}")
else:
    df_customers_enhanced.write.format("delta") \
        .mode("append") \
        .saveAsTable(customers_table)
    print(f"✅ Created {customers_table} with {customer_row_count} customers")

# COMMAND ----------
# MAGIC %md
# MAGIC ### Output Metrics for Downstream Tasks

# COMMAND ----------
# Return values for job orchestration
dbutils.notebook.exit(f""{{
    "status": "SUCCESS",
    "products_row_count": {product_row_count},
    "customers_row_count": {customer_row_count},
    "total_null_count": {product_null_count + customer_null_count},
    "execution_time": "{datetime.datetime.now().isoformat()}"
}}"")

2️⃣ Task 2: Data Quality Check (Condition Task)

# notebooks/02_data_quality_check.py

# Databricks notebook source
# MAGIC %md
# MAGIC ## Data Quality Validation
# MAGIC **Condition task to validate Bronze layer data quality**

# COMMAND ----------
import json

# Get upstream task results
upstream_result = dbutils.widgets.get("bronze_result")
result_data = json.loads(upstream_result)

products_count = result_data.get("products_row_count", 0)
customers_count = result_data.get("customers_row_count", 0)
null_count = result_data.get("total_null_count", 0)

print(f"Products: {products_count}, Customers: {customers_count}, Nulls: {null_count}")

# COMMAND ----------
# Quality rules (align with your data)
MIN_PRODUCTS = 90  # Your CSV has 100 products
MIN_CUSTOMERS = 1000  # Adjust based on your data
MAX_NULL_RATIO = 0.05  # 5% max nulls

total_records = products_count + customers_count
null_ratio = null_count / total_records if total_records > 0 else 1

quality_passed = (
    products_count >= MIN_PRODUCTS and
    customers_count >= MIN_CUSTOMERS and
    null_ratio <= MAX_NULL_RATIO
)

# COMMAND ----------
# Output for condition evaluation
result = {
    "quality_passed": quality_passed,
    "products_count": products_count,
    "customers_count": customers_count,
    "null_ratio": null_ratio,
    "quality_rules": {
        "min_products": MIN_PRODUCTS,
        "min_customers": MIN_CUSTOMERS,
        "max_null_ratio": MAX_NULL_RATIO
    }
}

print(f"Quality check: {'✅ PASSED' if quality_passed else '❌ FAILED'}")
dbutils.notebook.exit(json.dumps(result))

# pipelines/silver_declarative.yml
# Lakeflow Spark Declarative Pipeline definition

target: databricks_clar.silver
configuration:
  bundle.sourcePath: /Workspace/Users/your_email@company.com/medallion_pipeline

resources:
  - name: customers_enriched
    type: STREAMING_TABLE
    comment: "Silver layer customer data with enrichments"

    definition: |
      SELECT 
        customer_id,
        name,
        email,
        domain,
        location,
        signup_date,
        bronze_ingestion_timestamp,
        CASE 
          WHEN domain IN ('gmail.com', 'yahoo.com', 'hotmail.com') THEN 'Personal'
          ELSE 'Business'
        END as customer_type,
        datediff(current_date(), signup_date) as days_since_signup,
        source_hash
      FROM STREAM(databricks_clar.bronze.customers_bronze)
      WHERE email IS NOT NULL
        AND customer_id IS NOT NULL

  - name: products_enriched  
    type: STREAMING_TABLE
    comment: "Silver layer product data with category analysis"

    definition: |
      SELECT 
        product_id,
        product_name,
        category,
        price,
        source_process_date,
        bronze_ingestion_timestamp,
        CASE 
          WHEN price < 100 THEN 'Low'
          WHEN price < 300 THEN 'Medium'
          ELSE 'High'
        END as price_tier,
        round(price * 1.2, 2) as price_with_margin,
        source_hash
      FROM STREAM(databricks_clar.bronze.products_bronze)
      WHERE price > 0
        AND product_id IS NOT NULL

  - name: category_summary
    type: MATERIALIZED_VIEW  
    comment: "Pre-aggregated category metrics for Gold layer"

    definition: |
      SELECT 
        category,
        count(*) as product_count,
        avg(price) as avg_price,
        min(price) as min_price,
        max(price) as max_price,
        sum(price) as total_value,
        current_timestamp() as last_updated
      FROM databricks_clar.silver.products_enriched
      GROUP BY category

4️⃣ Task 4: Gold Aggregation (SQL Task)

-- notebooks/03_gold_aggregation.sql
-- Business intelligence aggregations

-- Create business metrics tables
CREATE OR REPLACE TABLE databricks_clar.gold.product_analytics AS
SELECT 
  category,
  price_tier,
  COUNT(*) as product_count,
  ROUND(AVG(price), 2) as avg_price,
  ROUND(SUM(price), 2) as total_value,
  ROUND(AVG(price_with_margin), 2) as avg_margin_price
FROM databricks_clar.silver.products_enriched
GROUP BY category, price_tier
ORDER BY category, price_tier;

-- Customer segmentation analysis  
CREATE OR REPLACE TABLE databricks_clar.gold.customer_segments AS
SELECT 
  customer_type,
  location,
  COUNT(*) as customer_count,
  AVG(days_since_signup) as avg_days_active,
  COUNT(DISTINCT domain) as unique_domains
FROM databricks_clar.silver.customers_enriched  
GROUP BY customer_type, location
ORDER BY customer_count DESC;

-- Category performance (based on your original analysis)
CREATE OR REPLACE TABLE databricks_clar.gold.category_performance AS
SELECT 
  category,
  product_count,
  avg_price,
  total_value,
  RANK() OVER (ORDER BY avg_price DESC) as price_rank,
  RANK() OVER (ORDER BY product_count DESC) as volume_rank,
  last_updated
FROM databricks_clar.silver.category_summary;

    variables:
      catalog_name: databricks_clar_prod

  # The Declarative Pipeline
  pipelines:
    silver_pipeline:
      name: medallion_silver_pipeline
      target: ${var.catalog_name}.${var.schema_silver}
      libraries:
        - file: 
            path: ./pipelines/silver_declarative.yml
      clusters:
        - num_workers: 2
          spark_version: "14.3.x-scala2.12"
          node_type_id: "i3.xlarge"
      serverless: true

  # The orchestration job
  jobs:
    medallion_job:
      name: medallion_pipeline_job

      # Schedule - daily at 2 AM EST  
      schedule:
        quartz_cron_expression: "0 0 2 * * ?"
        timezone_id: "America/New_York"
        pause_status: UNPAUSED

      # Notifications
      email_notifications:
        on_start: 
          - clarence@company.com
        on_failure:
          - clarence@company.com
          - data-team@company.com
        on_success:
          - clarence@company.com

      # The DAG tasks
      tasks:
        # Task 1: Bronze ingestion (your PySpark notebook)
        - task_key: bronze_ingestion
          notebook_task:
            notebook_path: ./notebooks/01_bronze_ingestion.py
            base_parameters:
              catalog_name: ${var.catalog_name}
              schema_name: ${var.schema_bronze}
              source_path: "/Volumes/${var.catalog_name}/${var.schema_bronze}/bronze_volume/"
          timeout_seconds: 1800  # 30 minutes
          max_retries: 1

        # Task 2: Quality validation  
        - task_key: data_quality_check
          depends_on:
            - task_key: bronze_ingestion
          condition_task:
            op: "GREATER_THAN" 
            left: "{{tasks.bronze_ingestion.values.total_null_count}}"
            right: "10"

        # Task 3: Silver processing (only if quality passes)
        - task_key: silver_processing
          depends_on:
            - task_key: data_quality_check  
              outcome: "false"  # null count <= 10
          pipeline_task:
            pipeline_id: ${resources.pipelines.silver_pipeline.id}
            full_refresh: false

        # Task 4: Gold aggregation
        - task_key: gold_aggregation
          depends_on:
            - task_key: silver_processing
          sql_task:
            file:
              path: ./notebooks/03_gold_aggregation.sql
            warehouse_id: "your_sql_warehouse_id"

        # Task 5: Alert handler (if quality fails)
        - task_key: quality_alert
          depends_on:
            - task_key: data_quality_check
              outcome: "true"  # null count > 10 
          notebook_task:
            notebook_path: ./notebooks/04_alert_handler.py
            base_parameters:
              quality_result: "{{tasks.data_quality_check.values.result}}"

      # Performance settings
      queue:
        enabled: true
      max_concurrent_runs: 1

🎯 Exam Key Points

LakeFlow Jobs Concepts Tested:

  1. Task Dependencies: Bronze → Quality Check → Silver → Gold (linear + conditional)
  2. Condition Tasks: Quality gate that routes to success or alert path
  3. Serverless Compute: Default for all task types (notebook, SQL, pipeline)
  4. Retries: Auto-optimization enabled by default on serverless
  5. Asset Bundles: Infrastructure as Code for CI/CD
  6. System Tables: Monitoring job performance and costs
  7. Triggers: Cron schedule (daily 2 AM) vs. continuous vs. file arrival
  8. Task Types: Notebook, Pipeline, SQL, Condition - each serves a purpose

Common Exam Traps:

  • ❌ "Use all-purpose compute for production jobs" → Use serverless/jobs compute
  • ❌ "Condition task outcome 'true' means success" → 'true' means condition was met (could be failure case)
  • ❌ "All downstream tasks run regardless of upstream status" → Failed tasks skip downstream unless configured otherwise
  • ❌ "Job parameters and task parameters are the same" → Job params cascade, task params are scoped
