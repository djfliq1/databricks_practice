02 — Table Types
🌊
Streaming Table @dp.table 
# Continuous updates, append-only, stateful
# Bronze / Silver layers
@dp.table(name='sales_stg')
def sales_stg():
    return spark.readStream\
        .option("skipChangeCommits", True)\
        .table("databricks_clar.silver.sales_enriched")

# skipChangeCommits=True needed when source
# table has updates/deletes (merge pattern)
📦
Materialized View @dp.materialized_view 
# Batch processing, full refresh, aggregations
# Gold layer / BI-ready
@dp.materialized_view(name='sales_trn')
def sales_trn():
    df = spark.read.table("sales_stg")
    df = df.withColumn(
        "priceAfterDiscount",
        col("total_amount") - col("discount")
    )
    return df
👁
Temp View — Batch @dp.view 
# Intermediate querying, NOT persisted
# Pipeline-scope only
@dp.view(name='demo_batch_view')
def demo_batch_view():
    df = spark.read.table("databricks_clar.silver.sales_enriched")
    return df
🔄
Temp View — Streaming @dp.view 
# Same decorator — reader determines behavior
# Stateful, micro-batch, not stored to disk
@dp.view(name='demo_stream_view')
def demo_stream_view():
    df = spark.readStream.table("databricks_clar.silver.sales_enriched")
    return df

05 — Append Flow
➕
Append Flow — Multi-Source + Backfill 
# STEP 1 — Create the empty target bucket first
dp.create_streaming_table(name="append_table")

# STEP 2 — Flow 1 (CSV from Volume folder)
@dp.append_flow(target="append_table", name="flow_1")
def flow_1():
    return spark.readStream.format("cloudFiles")\
        .option("cloudFiles.format", "csv")\
        .option("header", "true")\
        .option("inferSchema", "true")\
        .load("/Volumes/databricks_clar/bronze/autovol/raw/flow_1/")

# STEP 3 — Flow 2 (second source into same target)
@dp.append_flow(target="append_table", name="flow_2")
def flow_2():
    return spark.readStream.format("cloudFiles")\
        .option("cloudFiles.format", "csv")\
        .option("header", "true")\
        .option("inferSchema", "true")\
        .load("/Volumes/databricks_clar/bronze/autovol/raw/flow_2/")

06 — Auto CDC Flow (SCD)
🔁
Auto CDC Flow — SCD Type 1 + Type 2 replaces apply_changes() 
from pyspark import pipelines as dp
from pyspark.sql.functions import col

# STAGING TABLE — raw source ingest
@dp.table(name="scd_stg")
def scd_stg():
    return spark.readStream.table("databricks_clar.bronze.source")

### SCD TYPE 1 — overwrite, latest state only ###
dp.create_streaming_table(name="scd1_table")
dp.create_auto_cdc_flow(
    target             = "scd1_table",
    source             = "scd_stg",
    keys               = ["product_id"],
    sequence_by        = col("processDate"),    # col() required — not a string
    stored_as_scd_type = 1
)

### SCD TYPE 2 — keep full history ###
dp.create_streaming_table(name="scd2_table")
dp.create_auto_cdc_flow(
    target             = "scd2_table",
    source             = "scd_stg",
    keys               = ["product_id"],
    sequence_by        = col("processDate"),
    stored_as_scd_type = 2,
    except_column_list = ["processDate"]     # NOT except_columns
)
